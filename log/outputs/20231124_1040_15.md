```python
import re

# テキストから不要な文字を削除し、カンマを中点に置換する関数を再定義する
def clean_text(text):
    # シングルクォートを削除する
    text = re.sub(r"'", "", text)
    # リストの開始のブラケットを削除する
    text = re.sub(r"\['", "[", text)
    # リストの終了のブラケットを削除する
    text = re.sub(r"'\]", "]", text)
    # カンマを中点に置換する
    text = re.sub(r", ", "・", text)
    return text
```
