```python
import re

# テーブルの列幅を調整するための正規表現パターンを定義する
column_width_pattern = re.compile(r'^\|:--+\|--+:')

# テーブルの列幅を調整する関数を定義する
def adjust_column_width(text):
    # テーブルの2行目を置換する
    lines = text.split('\n')
    lines[1] = '|---|---|----------|----------|----------|'
    return '\n'.join(lines)

# テスト用のテキストを定義する
test_text = """
| クエリID   |   検索数k | matched_source                                                                                                                                                                                                                 | missed_source                                                                                                                                                                                                      | excessive_source                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                   | モデル     |
|:-----------|----------:|:-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------|:-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------|:---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------|:-----------|
"""

# テスト用のテキストの列幅を調整する
adjusted_text = adjust_column_width(test_text)

# 調整結果を表示する
print(adjusted_text)
```
