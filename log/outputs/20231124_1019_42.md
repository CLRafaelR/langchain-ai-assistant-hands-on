```python
import re

# テーブルの列幅を調整するための正規表現パターンを定義する
column_pattern = re.compile(r'(:?-{3,})')

# テーブルの列幅を調整する関数を定義する
def adjust_column_width(text):
    # 列幅を調整する
    return column_pattern.sub(lambda m: '---' if m.start() < text.find('|', text.find('|')+1) else '----------', text)

# テスト用のテキストを定義する
test_text = "|:-----------|----------:|:-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------|"

# テスト用のテキストの列幅を調整する
adjusted_text = adjust_column_width(test_text)

# 調整結果を表示する
print(adjusted_text)
```
